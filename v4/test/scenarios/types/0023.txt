-- given a function type signature ensure that it reports an error if it is unable to the explicit signature
type List a = Cons a (List a)
            | Nil;

range : a -> a -> List(Integer);
range min max =
    if min > max then
        Nil
    else
        Cons min (range (min + 1) max);
