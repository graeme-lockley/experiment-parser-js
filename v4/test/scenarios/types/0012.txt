-- given relational comparison operators ensure that they unify with boolean
opEqual a b = a == b;

opNotEqual a b = a != b;

opLessEqual a b = a <= b;

opLess a b = a < b;

opGreaterEqual a b = a >= b;

opGreater a b = a > b;
-- type
opEqual : a1 -> a1 -> Boolean
opNotEqual : a1 -> a1 -> Boolean
opLessEqual : a1 -> a1 -> Boolean
opLess : a1 -> a1 -> Boolean
opGreaterEqual : a1 -> a1 -> Boolean
opGreater : a1 -> a1 -> Boolean
_$EXPR : Unit
